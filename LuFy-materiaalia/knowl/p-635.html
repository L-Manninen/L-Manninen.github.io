\begin{gather*} \vec{F}_1 \cdot \dee \vec{r}_1 = (t^2\hat{i}) \cdot (\hat{j})\dee t = 0 \\ \vec{F}_2 \cdot \dee \vec{r}_2 = (\hat{i} + 2t\hat{j}) \cdot (\dee t \hat{i}) = \dee t \end{gather*}
